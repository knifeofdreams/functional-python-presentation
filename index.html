FizzBuzz

Write a program that iterates through the numbers from 1 to 100 and:

  • For multiples of 3 print "Fizz"
  • For multiples of 5 print "Buzz"
  • For numbers which are multiples of both 3 and 5 print "FizzBuzz"
  • For others, print the number itself

def const(constant):
    return lambda number: constant

def concat(f, g):
    return lambda number: f(number) + g(number)

def take_first(n, list):
    return list[:n]

def multiples_of_3():
    return map(lambda i: i * 3, range(1, 11))

def multiples_of_5():
    return map(lambda i: i * 5, range(1, 11))

def substract(list_from, what_list):
    return list(set(list_from) - set(what_list))

def add(list_1, list_2):
    return list_1 + list_2

def repeat(value, times):
    return [value] * times

def for_all(list, function):
    return dict(zip(list, repeat(function, len(list))))

multiples_of_3 = take_first(4, multiples_of_3())
multiples_of_5 = take_first(2, multiples_of_5())
multiples_of_both = [0]
others = substract(range(1,15), add(multiples_of_3, multiples_of_5))

id = lambda number: str(number)
fizz = const("Fizz")
buzz = const("Buzz")
fizzbuzz = concat(fizz, buzz)

fizzbuzz_lambdas = dict(for_all(multiples_of_3, fizz).items() + for_all(multiples_of_5, buzz).items() + for_all(multiples_of_both, fizzbuzz).items() + for_all(others, id).items())
